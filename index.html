挂载数据卷 # 指定路径挂载 $ docker run -it -v <宿主机路径>:<容器内路径> nginx # 具名挂载，主机路径 /var/lib/docker/volumes/<卷名>/_data $ docker run -it -v <卷名>:<容器内路径> nginx # 匿名挂载，主机路径 /var/lib/docker/volumes/<一串随机字符>/_data $ docker run -it -v <容器内路径> nginx 查看数据卷 列出数据卷 $ docker volume ls DRIVER VOLUME NAME local 0b60738f27c8adc38559981ad727d36a722b70b1736b20d8f3020ffc88d202b2 local 21f61aa404a0fe9c15a6a6ad4a6a056a0ee009a98af593c2092a14062fa7c02e local test # 上面两个是匿名挂载 # test是具名挂载 查看卷详情 $ docker volume inspect <卷名> [ { "CreatedAt": "2022-01-24T17:08:58+08:00", "Driver": "local", "Labels": null, "Mountpoint": "/var/lib/docker/volumes/test/_data", "Name": "test", "Options": null, "Scope": "local" } ] 读写权限 # ro 只读，容器对该数据卷只读 $ docker run -d -v /home/nginx:/etc/nginx:ro nginx # rw 读写，容器对该数据卷可读写 $ docker run -d -v /home/nginx:/etc/nginx:rw nginx
